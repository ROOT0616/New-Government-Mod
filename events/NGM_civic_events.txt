###############################

namespace = ngm_civic

# 青年政治
country_event = {
	id = ngm_civic.1
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_civic = civic_fsass_youth_politics
	}
	
	immediate = {
		every_owned_species = {
			limit = {
				Not = {  has_species_flag = flag_ngm_got_civic_young }
			}
			set_species_flag = { 
				flag = flag_ngm_got_civic_young
			}
			add_trait = trait_ngm_youth_politics
		}
	}
}
# リーダーの特性付
country_event = {
	id = ngm_civic.2
	hide_window = no
	is_triggered_only = no

	trigger = {
		has_civic = civic_fsass_youth_politics
	}
	
	immediate = {
		every_leader = {
			limit = {
				Not = {  has_leader_flag = flag_ngm_got_civic_young_bonus }
			}
			set_leader_flag = { 
				flag = flag_ngm_got_civic_young_bonus
			}
				add_trait = leader_trait_adaptable
				add_experience = 650
		}
	}
}
