		}

	planet_modifier = {
		job_bureau_of_ceremonies_add = 2
	}

		trigger = {
			exists = owner
		}
		text = job_bureau_of_ceremonies_effect_desc
	}
	on_built = {
		owner = {
			if = {
				limit = {
					NOT = {
						has_country_flag = build_ngm_bureau_of_ceremonies
					}
				}
				set_country_flag = build_ngm_bureau_of_ceremonies
			}
		}
	}
	on_destroy = {
		owner = {
			remove_country_flag = build_ngm_bureau_of_ceremonies
		}
	}
}
