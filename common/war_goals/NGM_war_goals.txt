# ※聖戦（非精神主義国家に対して精神主義を強要する宣戦布告事由を得る）
wg_ngm_holy_wars = {
	casus_belli = cb_ngm_holy_wars

	hide_if_no_cb = yes
	
	cede_claims = no

	surrender_acceptance = -100

	war_exhaustion = 0.75
	
	potential = {
		always = yes
	}
	
	possible = {
		always = yes
	}
	
	allowed_peace_offers = {
		status_quo
		surrender
		demand_surrender
	}
	
	on_accept = {
		custom_tooltip = wg_force_ideology_accept
		custom_tooltip = wg_force_ideology_accept_2
		if = {
			limit = {
				from = {
					has_federation = yes
				}
			}
			custom_tooltip = wg_force_ideology_accept_3
		}
		if = {
			limit = { from = { has_been_declared_crisis = no } }
			from = {
				set_subject_of = {
					who = root
					subject_type = vassal
				}
			}
		}
		hidden_effect = {
			from = {
				# copy_ethos_and_authority = root
				if = {
					limit = { 
						has_ethic = ethic_materialist
					}
					country_remove_ethic = ethic_materialist
					country_add_ethic = ethic_spiritualist
				}
				if = {
					limit = { 
						has_ethic = ethic_fanatic_materialist
					}
					country_remove_ethic = ethic_fanatic_materialist
					country_add_ethic = ethic_fanatic_spiritualist
				}
				if = {
					limit = {
						has_civic = civic_galactic_sovereign
					}
					force_remove_civic = civic_galactic_sovereign
				}
				if = {
					limit = {
						has_civic = civic_galactic_sovereign_megacorp
					}
					force_remove_civic = civic_galactic_sovereign_megacorp
				}
			}
			from = {
				add_modifier = {
					modifier = recently_liberated
					days = 7300
				}
			}
			add_opinion_modifier = {
				modifier = opinion_we_liberated
				who = from
			}
			from = {
				add_opinion_modifier = {
					modifier = opinion_liberated_us
					who = root
				}
			}
			end_rivalry = from
			from = {
				end_rivalry = root
			}
			every_system_within_border = {
				limit = {
					from = {
						has_claim = prev
					}
				}
				remove_claims = {
					who = from
				}
			}
			every_country = {
				limit = {
					OR = {
						is_in_federation_with = root
						AND = {
							is_subject = yes
							exists = overlord
							overlord = {
								is_same_value = root
							}
						}
					}
				}
				end_rivalry = from
				from = {
					end_rivalry = prev
				}
				add_opinion_modifier = {
					modifier = opinion_we_liberated
					who = from
				}
				from = {
					add_opinion_modifier = {
						modifier = opinion_liberated_us
						who = prev
					}
				}
				every_system_within_border = {
					limit = {
						from = {
							has_claim = prev
						}
					}
					remove_claims = {
						who = from
					}
				}
			}
			from = {
				add_threat = {
					who = root
					amount = 0.1
				}
			}
			from = {
				leave_alliance = {
					override_requirements = yes
				}
			}
		}
	}
	ai_weight = {
		weight = 100
	}
}
